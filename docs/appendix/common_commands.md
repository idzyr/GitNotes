# Git常用命令

```bash
pwd # 查看当前工作目录
mkdir # 新建目录
cat 文件名 # 查看文件内的内容。
touch [文件名加后缀] #新建文件
ls -a # 显示所有目录
rm 文件名 # 删除文件。这样删除git不会记录
mv [源文件名] [新文件名]  # 更改文件名
vim 文件名 # 查看文件（退出按大写Z两次）
> # 添加内容到文本中（覆盖方式）
>> #添加内容到文本中（追加方式）


# git相关命令。
git init # 初始化一个.git仓库
git status # 查看文件状态(查看文件是工作区或暂存区)

# 文件添加到暂存区命令合集。
git add 文件名/目录 # 添加文件到暂存区 同时添加多个文件可以使用空格分割文件名
git add --all # 将当前目录所有文件添加到暂存区
git add .
git add -A  
# 以上两个 add命令会把工作区文件全部放到暂存区。

git commit -m “描述” # 添加文件到本地git库 【本地仓库=历史记录区】
git commit -a -m “信息” # 自动添加到暂存区并提交

# 版本提交log
git log # 查看提交日志 ，从最近提交~早期提提交。
git log --pretty=oneline # 简便显示，只显示版本号，和提交描述内容。展示方式同上。个命令，会显示另一个分支的日志。

git rm 文件名 # 从暂存区和工作区删除文件，git会记录删除操作
git rm --cached 文件名 # 只从暂存区删除文件/目录，git会记录。
git clone "项目url" #通过http克隆项目。或仓库ssh地址
git mv 源文件名 [路径，不设置路径默认当前路径]新文件名 # 从工作区重命名文件/目录/或移动文件

git diff 文件 # 查看文件修改了那些内容。 只要当文件内容有变更时才会展示修改的内容。工作区文件与暂存区对比。
git diff HEAD -- 文件 # 查看工作区与版本库里最新文件的不同之处。没有不同不会显示。

# 暂存区文件恢复
git restore --staged <file> # 作用是将暂存区的文件从暂存区撤出，但不会更改文件
```
